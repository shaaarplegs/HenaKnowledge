package moee.henaknowledge.repository;

import moee.henaknowledge.dal_interfaces.IAdminDAL;
import moee.henaknowledge.module.Admin;

import java.sql.Date;
import java.util.ArrayList;
import java.util.List;

public class AdminDalStub implements IAdminDAL {


    static List<Admin> admins;

    public AdminDalStub()
    {
        admins = new ArrayList<>();
    }


    @Override
    public Admin getAdminByUsername(String username) {
        for (Admin a: admins) {
            if(a.getUsername().equals(username)) {
                return a;
            }
        }
        return null;
    }

    @Override
    public Admin getAdminByPersonID(int ID) {
        for (Admin a: admins) {
            if(a.getPersonID() == ID) {
                return a;
            }
        }
        return null;
    }

    @Override
    public List<Admin> getAllAdmins() {
        return admins;
    }

    @Override
    public void AddAdmin(Admin admin) {
        admins.add(admin);
    }

    @Override
    public void DeleteAdmin(int ID) {
        for (Admin a: admins) {
            if(a.getPersonID() == ID) {
                admins.remove(a);
            }
        }
    }

    @Override
    public void setAdminInfoById(int ID, String firstname, String lastname, Date DateOfBirth, String Email) {
        for (Admin a: admins) {
            if(a.getPersonID() == ID) {
                a.setFirstName(firstname);
                a.setLastName(lastname);
                a.setDateOfBirth(DateOfBirth);
                a.setEmail(Email);
            }
        }
    }
}
